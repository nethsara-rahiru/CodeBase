PHP Beginner Cheat Sheet

PHP Beginner Cheat Sheet

1. PHP Basics

<?php
// Comments
echo "Hello World";
$name = "Nethsara";
$age = 20;

// Concatenation
echo "Name: " . $name;
?>

2. Operators


// Arithmetic
+ - * / %
// Assignment
= += -= *= /=
// Comparison
== != === !== < > <= >=
// Logical
&& || !

3. Strings


$str = "Hello World";
echo strlen($str);     					// length
echo strtoupper($str); 					// UPPERCASE
echo strtolower($str); 					// lowercase
echo str_replace("World","PHP",$str); 	// Replace

4. Arrays


$fruits = ["Apple","Banana","Orange"];
$person = ["name" => "Nethsara","age" => 20];

5. Conditionals


if ($age > 18) { echo "Adult"; } 
elseif ($age == 18) { echo "Just Adult"; } 
else { echo "Minor"; }

echo ($age >= 18) ? "Adult" : "Minor";

6. Loops


for ($i = 0; $i < 5; $i++) { echo $i; }

$i = 0;
while ($i < 5) { echo $i; $i++; }

foreach ($fruits as $fruit) { echo $fruit; }

7. Functions


function greet($name) { return "Hello $name"; }
echo greet("Nethsara");

8. Superglobals


$_POST['fieldname'];
$_GET['fieldname'];
$_SERVER['PHP_SELF'];
$_COOKIE['name'];
$_SESSION['name'];

9. Forms


<form method="POST">
    <input name="username">
    <input type="password" name="password">
    <button>Submit</button>
</form>

<?php
if ($_POST) {
    $user = $_POST['username'];
    $pass = $_POST['password'];
    echo "Username: $user";

?>

10. Passwords


$hash = password_hash($password, PASSWORD_DEFAULT);
if (password_verify($password, $hash)) { echo "Correct Password"; }
else { echo "Wrong Password"; }

11. PDO

<?php
$dsn = "mysql:host=localhost;dbname=testdb";
$con = new PDO($dsn, "root", "");

// Select
$stmt = $con->prepare('SELECT * FROM users WHERE phone = :phone');
$stmt->execute([":phone" => $phone]);
$row = $stmt->fetch(PDO::FETCH_ASSOC);

// Insert
$stmt = $con->prepare('INSERT INTO users(name, phone) VALUES(:name, :phone)');
$stmt->execute([":name" => $name, ":phone" => $phone]);
?>

12. Common Errors

• Missing semicolon ";"
• Wrong variable name
• Empty POST fields
• Using "password_hash()" during login instead of "password_verify()"
• Not using prepared statements (SQL injection risk)
